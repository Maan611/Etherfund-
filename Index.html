document.addEventListener("DOMContentLoaded", function () { const connectWalletBtn = document.getElementById("connectWallet"); const sendDonationBtn = document.getElementById("sendDonation"); const walletStatus = document.getElementById("walletStatus"); let userAccount; const donationAddress = "YOUR_ETH_WALLET_ADDRESS"; // Replace with your actual Ethereum address // Connect to MetaMask connectWalletBtn.addEventListener("click", async () => { if (window.ethereum) { try { const accounts = await window.ethereum.request({ method: "eth_requestAccounts" }); userAccount = accounts[0]; walletStatus.innerText = `Connected: ${userAccount}`; connectWalletBtn.innerText = "Connected"; } catch (error) { console.error(error); walletStatus.innerText = "Connection failed!"; } } else { walletStatus.innerText = "MetaMask not detected! Install MetaMask."; } }); // Send ETH Donation sendDonationBtn.addEventListener("click", async () => { if (!userAccount) { walletStatus.innerText = "Please connect MetaMask first!"; return; } let amount = document.getElementById("ethAmount").value; if (!amount || isNaN(amount) || amount <= 0) { walletStatus.innerText = "Enter a valid ETH amount!"; return; } try { const transaction = await window.ethereum.request({ method: "eth_sendTransaction", params: [{ from: userAccount, to: donationAddress, value: (amount * 1e18).toString(16) // Convert ETH to Wei (Hex format) }] }); walletStatus.innerText = `Donation sent! TX Hash: ${transaction}`; } catch (error) { console.error(error); walletStatus.innerText = "Transaction failed!"; } }); });
